```js
//_app.js
import { MDXProvider } from "@mdx-js/react";
import CodeBlock from "../components/codeblock";

const components = {
  code: CodeBlock,
};

export default function App({ Component, pageProps }) {
  return (
    <MDXProvider components={components}>
      <Component {...pageProps} />
    </MDXProvider>
  );
}
```
